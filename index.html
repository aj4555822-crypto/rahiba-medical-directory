        function updateStats() {

            const clinics = medicalData.filter(item => item.type === 'clinic').length;

        }

        // دالة الفلترة والبحث الفوري الفعال
        function filterData() {
            const q = document.getElementById('searchInput').value.toLowerCase().trim();
            const filtered = medicalData.filter(item =>
                item.name.toLowerCase().includes(q) ||
                item.specialty.toLowerCase().includes(q) ||
                item.address.toLowerCase().includes(q)
            );
            renderDirectory(filtered);
            document.getElementById('noResults').classList.toggle('hidden', filtered.length > 0);
        }
